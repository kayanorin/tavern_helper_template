        // 排名项正则：捕获称号、姓名、所属、武学、评语
        const rankingItemRegex = /第[一二三四五六七八九十百千万]+位：【([^】]+)】([^\n]+)\n所属：([^\n]+)\n武学：([^\n]+)\n评语：([^\n]+)/g;
        
        // 获取内容
        const content = `$2`;
        const container = document.getElementById('rankingContent');
        
        // 解析并渲染排名项
        let match;
        while ((match = rankingItemRegex.exec(content)) !== null) {
            const itemEl = document.createElement('div');
            itemEl.className = 'ranking-item';
            
            const headerEl = document.createElement('div');
            headerEl.className = 'ranking-header';
            headerEl.innerHTML = `<span class="toggle-icon">▶</span>${match[0].split('\n')[0]}`;
            
            const detailsEl = document.createElement('div');
            detailsEl.className = 'ranking-details';
            detailsEl.innerHTML = `
                <div class="detail-line">
                    <span class="detail-label">所属：</span>
                    <span class="detail-content">${match[3]}</span>
                </div>
                <div class="detail-line">
                    <span class="detail-label">武学：</span>
                    <span class="detail-content">${match[4]}</span>
                </div>
                <div class="detail-line">
                    <span class="detail-label">评语：</span>
                    <span class="detail-content">${match[5]}</span>
                </div>
            `;
            
            headerEl.addEventListener('click', () => {
                itemEl.classList.toggle('expanded');
            });
            
            itemEl.appendChild(headerEl);
            itemEl.appendChild(detailsEl);
            container.appendChild(itemEl);
        }
    </script>
